         ```typescript

             ],
           },
         };
         ```
         Note: The hover highlight opacity will be handled via feature-state in a later enhancement. For now use flat 0.6.

       - Define outline layer OUTSIDE the component:
         ```typescript
         const outlineLayer: LineLayer = {
           id: 'neighbourhood-outline',
           type: 'line',
           paint: {
             'line-color': '#ffffff',
             'line-width': 1,
             'line-opacity': 0.5,
           },
         };
         ```

       - Component implementation:
         - Call `useNeighbourhoods()` for geojson/loading
         - Call `useNeighbourhoodDetail()` to wire detail fetching on selection
         - Get setSelected, setHovered from useMapStore
         - Maintain local tooltip state: `useState<{x:number; y:number; name:string; vibe:string} | null>(null)`

         - handleHover callback (useCallback):
           ```typescript
           const feature = event.features?.[0];
           if (feature) {
             setTooltipInfo({ x: event.point.x, y: event.point.y, name: feature.properties.NEIGHBORHOOD_NAME, vibe: feature.properties.dominant_vibe });
             setHovered(feature.properties.NEIGHBORHOOD_NUMBER);
           } else {
             setTooltipInfo(null);
             setHovered(null);
           }
           ```

         - handleClick callback (useCallback):
           ```typescript
           const feature = event.features?.[0];
           if (feature) {
             setSelected(feature.properties.NEIGHBORHOOD_NUMBER);
           }
           ```

         - handleMouseLeave callback: setTooltipInfo(null); setHovered(null);

         - If loading or !geojson, return `<MapSkeleton />`

         - Render:
           ```jsx
           <div style={{ width: '100%', height: '100vh', position: 'relative' }}>
             <Map
               initialViewState={INITIAL_VIEW}
               style={{ width: '100%', height: '100%' }}
               mapStyle={BASEMAP_STYLE}
               interactiveLayerIds={['neighbourhood-fill']}
               onMouseMove={handleHover}
               onMouseLeave={handleMouseLeave}
               onClick={handleClick}
             >
               <Source id="neighbourhoods" type="geojson" data={geojson}>
                 <Layer {...fillLayer} />
                 <Layer {...outlineLayer} />
               </Source>
             </Map>
             {tooltipInfo && <Tooltip {...tooltipInfo} />}
           </div>
           ```

       - CRITICAL: pass `interactiveLayerIds={['neighbourhood-fill']}` to <Map>. Without this, event.features will always be empty.
       - CRITICAL: import 'maplibre-gl/dist/maplibre-gl.css' is already in main.tsx, do NOT import again here.
       - CRITICAL: Define fillLayer and outlineLayer as module-level constants to prevent MapLibre re-paints.
