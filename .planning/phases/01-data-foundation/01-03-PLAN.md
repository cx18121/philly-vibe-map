```sql
CREATE TABLE IF NOT EXISTS businesses (
    business_id   TEXT PRIMARY KEY,
    name          TEXT NOT NULL,
    latitude      REAL,
    longitude     REAL,
    neighbourhood_id   TEXT,   -- NTACode; NULL if outside Manhattan+Brooklyn

    city          TEXT,
    state         TEXT,
    attributes    TEXT         -- platform-specific fields as JSON blob (DATA-06)

    review_id     TEXT PRIMARY KEY,
    business_id   TEXT NOT NULL REFERENCES businesses(business_id),

    review_date   TEXT NOT NULL,
    source        TEXT NOT NULL DEFAULT 'yelp',
    useful        INTEGER,
    funny         INTEGER,
    cool          INTEGER,
    FOREIGN KEY (business_id) REFERENCES businesses(business_id)
);

CREATE INDEX IF NOT EXISTS idx_reviews_business ON reviews(business_id);
CREATE INDEX IF NOT EXISTS idx_reviews_date ON reviews(review_date);
```

```python
02_build_schema.py
Create (or verify) the Phase 1 SQLite schema.

Usage:

from __future__ import annotations

import argparse
import datetime
import sqlite3
import sys
from pathlib import Path

_ANSI = {"WARN": "\033[93m", "FAIL": "\033[91m", "INFO": ""}
_RESET = "\033[0m"

def _log(level: str, msg: str) -> None:
    ts = datetime.datetime.utcnow().strftime("%Y-%m-%d %H:%M:%S")
    color = _ANSI.get(level, "") if sys.stdout.isatty() else ""
    reset = _RESET if color else ""
    print(f"{color}[{level}] {ts}  {msg}{reset}", flush=True)


CREATE_BUSINESSES = """
CREATE TABLE IF NOT EXISTS businesses (
    business_id        TEXT PRIMARY KEY,
    name               TEXT NOT NULL,
    latitude           REAL,
    longitude          REAL,
    neighbourhood_id   TEXT,
    neighbourhood_name TEXT,
    city               TEXT,
    state              TEXT,
    attributes         TEXT
);
"""

CREATE_REVIEWS = """
CREATE TABLE IF NOT EXISTS reviews (
    review_id   TEXT PRIMARY KEY,
    business_id TEXT NOT NULL,
    text        TEXT NOT NULL,
    stars       INTEGER NOT NULL,
    review_date TEXT NOT NULL,
    source      TEXT NOT NULL DEFAULT 'yelp',
    useful      INTEGER,
    funny       INTEGER,
    cool        INTEGER,
    FOREIGN KEY (business_id) REFERENCES businesses(business_id)
);
"""

CREATE_INDICES = """
CREATE INDEX IF NOT EXISTS idx_reviews_business    ON reviews(business_id);
CREATE INDEX IF NOT EXISTS idx_reviews_date        ON reviews(review_date);
CREATE INDEX IF NOT EXISTS idx_businesses_neighbourhood ON businesses(neighbourhood_id);
"""


def build_schema(db_path: str) -> None:
    """Create (or verify) Phase 1 schema in the SQLite database at db_path."""
    path = Path(db_path)
    path.parent.mkdir(parents=True, exist_ok=True)

    conn = sqlite3.connect(db_path)
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA synchronous=NORMAL")
    conn.execute("PRAGMA foreign_keys=ON")

    conn.executescript(CREATE_BUSINESSES)
    conn.executescript(CREATE_REVIEWS)
    conn.executescript(CREATE_INDICES)
    conn.commit()
    conn.close()
    _log("INFO", f"Schema created/verified at {db_path}")


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--db", default="data/output/reviews.db")
    args = parser.parse_args()
    try:
        build_schema(args.db)
    except Exception as exc:
        _log("FAIL", f"Database write failed: {exc}. Check disk space and file permissions at {args.db}.")
        sys.exit(1)
```

Create scripts/03_assign_neighbourhoods.py:

The script performs two passes:
1. First pass (streaming): read business NDJSON, filter to rows where lat/lng are not null and lat/lng fall within approximate NYC bounding box (-74.05 to -73.70 lon, 40.57 to 40.92 lat). Accumulate as a list of dicts.
2. Spatial join (batch): build GeoDataFrame from list, run sjoin against NTA polygons, assign neighbourhood_id and neighbourhood_name.
3. Write (batch): INSERT OR IGNORE all businesses (including those outside NTAs — they get NULL neighbourhood_id).

```python
"""
03_assign_neighbourhoods.py
Stream Yelp business NDJSON, assign each NYC business to an NTA via GeoPandas sjoin,
write results to the businesses table in reviews.db.

Usage:
    YELP_DATA_DIR=/path/to/yelp python scripts/03_assign_neighbourhoods.py [--db data/output/reviews.db]
"""
from __future__ import annotations

import argparse
import datetime
import json
import os
import sqlite3
import sys
from pathlib import Path

import geopandas as gpd
import orjson
from shapely.geometry import box

NTA_PATH = "data/boundaries/nta_2020_manhattan_brooklyn.geojson"
CURATION_PATH = "data/boundaries/nta_name_curation.json"

# Bounding box slightly larger than NTA extent — pre-filter before sjoin
NYC_BBOX = box(-74.05, 40.57, -73.70, 40.92)

BATCH_SIZE = 10_000
WARN_SUPPRESS_THRESHOLD = 1_000

_ANSI = {"WARN": "\033[93m", "FAIL": "\033[91m", "INFO": ""}
_RESET = "\033[0m"

def _log(level: str, msg: str) -> None:
    ts = datetime.datetime.utcnow().strftime("%Y-%m-%d %H:%M:%S")
    color = _ANSI.get(level, "") if sys.stdout.isatty() else ""
    reset = _RESET if color else ""
    print(f"{color}[{level}] {ts}  {msg}{reset}", flush=True)


def assign_neighbourhoods(business_path: str, db_path: str) -> dict:
    """Stream business NDJSON, spatially join to NTA, write to SQLite businesses table.

    Args:
        business_path: Path to yelp_academic_dataset_business.json
        db_path: Path to reviews.db (schema must already exist)

    Returns:
        dict: {total_scanned, nyc_candidates, assigned, missing_coords, outside_nta}
    """
    # Load NTA boundaries and name curation
    nta_gdf = gpd.read_file(NTA_PATH)
    assert nta_gdf.crs is not None and nta_gdf.crs.to_epsg() == 4326, (
        f"NTA GeoDataFrame CRS must be EPSG:4326, got {nta_gdf.crs}"
    )

    with open(CURATION_PATH) as f:
        curation = json.load(f)

    # First pass: stream businesses, pre-filter to NYC bbox
    _log("INFO", "Yelp business file open — streaming parse started")
    total_scanned = 0
    missing_coords = 0
    warn_count = 0
    nyc_records = []  # dicts for GeoDataFrame construction

    with open(business_path, "rb") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            total_scanned += 1
            record = orjson.loads(line)
            lat = record.get("latitude")
            lon = record.get("longitude")
            if lat is None or lon is None:
                missing_coords += 1
                if missing_coords <= WARN_SUPPRESS_THRESHOLD:
                    _log("WARN", f"Skipped business {record.get('business_id', '?')}: missing lat/lng")
                continue
            from shapely.geometry import Point
            pt = Point(float(lon), float(lat))
            if NYC_BBOX.contains(pt):
                nyc_records.append({
                    "business_id": record["business_id"],
                    "name": record.get("name", ""),
                    "latitude": float(lat),
                    "longitude": float(lon),
                    "city": record.get("city"),
                    "state": record.get("state"),
                    "attributes": json.dumps(record.get("attributes") or {}),
                    "_geometry": pt,
                })

    if missing_coords > WARN_SUPPRESS_THRESHOLD:
        _log("WARN", f"{missing_coords} records skipped for missing lat/lng (suppressed individual warnings)")

    _log("INFO", f"NYC businesses identified: {len(nyc_records)} records")

    if not nyc_records:
        _log("WARN", "No NYC businesses found — businesses table will be empty. Review probe results.")
        return {
            "total_scanned": total_scanned,
            "nyc_candidates": 0,
            "assigned": 0,
            "missing_coords": missing_coords,
            "outside_nta": 0,
        }

    # Spatial join
    import pandas as pd
    df = pd.DataFrame(nyc_records)
    gdf_biz = gpd.GeoDataFrame(df, geometry="_geometry", crs="EPSG:4326")
    assert gdf_biz.crs.to_epsg() == 4326  # both sides must be EPSG:4326

    joined = gpd.sjoin(
        gdf_biz,
        nta_gdf[["NTACode", "NTAName", "geometry"]],
        how="left",
        predicate="within",
    )
    _log("INFO", f"Spatial join complete — {joined['NTACode'].notna().sum()} businesses assigned to neighbourhoods")

    # Write to SQLite
    conn = sqlite3.connect(db_path)
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA synchronous=NORMAL")

    batch = []
    assigned = 0
    outside_nta = 0

    for _, row in joined.iterrows():
        nta_code = row.get("NTACode")
        nta_name = None
        if pd.notna(nta_code):
            nta_name = curation.get(str(nta_code), row.get("NTAName"))
            assigned += 1
        else:
            outside_nta += 1

        batch.append((
            row["business_id"],
            row["name"],
            row["latitude"],
            row["longitude"],
            nta_code if pd.notna(nta_code) else None,
            nta_name,
            row.get("city"),
            row.get("state"),
            row["attributes"],
        ))

        if len(batch) >= BATCH_SIZE:
            conn.executemany(
                "INSERT OR IGNORE INTO businesses "
                "(business_id, name, latitude, longitude, neighbourhood_id, neighbourhood_name, city, state, attributes) "
                "VALUES (?,?,?,?,?,?,?,?,?)",
                batch,
            )
            conn.commit()
            batch.clear()

    if batch:
        conn.executemany(
            "INSERT OR IGNORE INTO businesses "
            "(business_id, name, latitude, longitude, neighbourhood_id, neighbourhood_name, city, state, attributes) "
            "VALUES (?,?,?,?,?,?,?,?,?)",
            batch,
        )
        conn.commit()

    conn.close()
    return {
        "total_scanned": total_scanned,
        "nyc_candidates": len(nyc_records),
        "assigned": assigned,
        "missing_coords": missing_coords,
        "outside_nta": outside_nta,
    }


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--db", default="data/output/reviews.db")
    args = parser.parse_args()

    data_dir = os.environ.get("YELP_DATA_DIR")
    if not data_dir:
        _log("FAIL", "YELP_DATA_DIR not set. Export it to the directory containing the Yelp Open Dataset files.")
        sys.exit(1)

    business_file = Path(data_dir) / "yelp_academic_dataset_business.json"
    if not business_file.exists():
        _log("FAIL", f"Dataset file not found at {business_file}.")
        sys.exit(1)

    try:
        result = assign_neighbourhoods(str(business_file), args.db)
        _log("INFO", f"Import complete — {result['assigned']} businesses assigned to neighbourhoods")
        _log("INFO", f"Outside NTA polygons (no neighbourhood): {result['outside_nta']}")
    except Exception as exc:
        _log("FAIL", f"Database write failed: {exc}. Check disk space and file permissions at {args.db}.")
        sys.exit(1)
```
